Q1)
SELECT lon,lat FROM adresse
WHERE numero = 19 AND code_postal = 91120 AND nom_commune = "Palaiseau" AND nom_voie = "Place Marguerite Perey"

Q2)  
SELECT MAX(lat), nom_commune FROM adresse 
GROUP BY nom_commune

Q3)

SELECT nom_commune FROM adresse
GROUP BY nom_voie
HAVING COUNT(distinct code_postal) >= 2


Q4)
SELECT nom_commune FROM adresse 
GROUP BY nom_commune
HAVING min(lat) > 
(SELECT MAX(lat) FROM adresse WHERE nom_commune = "Palaiseau")
